          import { cn } from "@/lib/utils";

          function useMousePosition(proximityRange) {
            const wrapperRef = useRef(null);
            const [mouseState, setMouseState] = React.useState({
              mousePositionX: 0,
              mousePositionY: 0,
              isWithinRange: false,
            });

            const handlePointerMovement = useCallback(
              (event) => {
                if (!wrapperRef.current) return;

                const bounds = wrapperRef.current.getBoundingClientRect();
                const { clientX, clientY } = event;

                const isInProximity =
                  clientX >= bounds.left - proximityRange &&
                  clientX <= bounds.right + proximityRange &&
                  clientY >= bounds.top - proximityRange &&
                  clientY <= bounds.bottom + proximityRange;

                setMouseState({
                  mousePositionX: clientX,
                  mousePositionY: clientY,
                  isWithinRange: isInProximity,
                });
              },
              [proximityRange]
            );

            useEffect(() => {
              document.addEventListener("pointermove", handlePointerMovement);
              return () =>
                document.removeEventListener("pointermove", handlePointerMovement);
            }, [handlePointerMovement]);

            return { wrapperRef, mouseState };
          }

            elementRef,
            globalMouseX,
            globalMouseY,
            isWithinRange,
            illuminationRadius
          ) {
            const localMouseX = useMotionValue(-illuminationRadius);
            const localMouseY = useMotionValue(-illuminationRadius);
            const [isCardActive, setIsCardActive] = React.useState(false);

            useEffect(() => {
              if (!elementRef.current || !isWithinRange) {
                setIsCardActive(false);
                localMouseX.set(-illuminationRadius);
                localMouseY.set(-illuminationRadius);
                return;
              }

              const rect = elementRef.current.getBoundingClientRect();
              const extendedProximity = 100;

              const isNearCard =
                globalMouseX >= rect.left - extendedProximity &&
                globalMouseX <= rect.right + extendedProximity &&
                globalMouseY >= rect.top - extendedProximity &&
                globalMouseY <= rect.bottom + extendedProximity;

              setIsCardActive(isNearCard);

              if (isNearCard) {
                localMouseX.set(globalMouseX - rect.left);
                localMouseY.set(globalMouseY - rect.top);
              } else {
                localMouseX.set(-illuminationRadius);
                localMouseY.set(-illuminationRadius);
              }
            }, [
              globalMouseX,
              globalMouseY,
              isWithinRange,
              illuminationRadius,
              localMouseX,
              localMouseY,
            ]);

            return { localMouseX, localMouseY, isCardActive };
          }

          export function CursorCardsContainer({
            children,
            className,
            proximityRange = 400,
          }) {
            const { wrapperRef, mouseState } = useMousePosition(proximityRange);

            const enhancedChildren = React.Children.map(children, (child) => {
              if (React.isValidElement(child) && child.type === CursorCard) {

                {enhancedChildren}
              </div>
            );
          }

            children,
            className,
            illuminationRadius = 200,
            illuminationColor = "#FFFFFF10",
            illuminationOpacity = 0.8,
            primaryHue = "#FC6756",
            secondaryHue = "#F8CF3E",

            globalMouseY = 0,
            isWithinRange = false,

              globalMouseX,
              globalMouseY,
              isWithinRange,
              illuminationRadius
            );

              radial-gradient(${illuminationRadius}px circle at ${localMouseX}px ${localMouseY}px,

              ${secondaryHue},
              ${borderColor} 100%
              )
            `;

              radial-gradient(${illuminationRadius}px circle at ${localMouseX}px ${localMouseY}px, 
              ${illuminationColor}, transparent 100%)
            `;

            return (
              <div
                ref={elementRef}
                className={cn("group relative rounded-[inherit]", className)}
              >
                <motion.div
                  className="pointer-events-none absolute inset-0 rounded-[inherit]"
                  style={{ background: gradientBackground }}
                />
                <div className="absolute inset-px rounded-[inherit] bg-white dark:bg-black" />

                  className={cn(

          import React, { useCallback, useEffect, useRef } from "react";

            className?: string;

            illuminationColor?: string;

            secondaryHue?: string;

            globalMouseX?: number;
            globalMouseY?: number;
            isWithinRange?: boolean;

              mousePositionX: 0,
              mousePositionY: 0,

              (event: PointerEvent) => {

```

<CursorCardsContainer>

    <div className="h-full w-full py-4 px-6 max-w-72 space-y-2">
      <h3 className="font-gilroy text-2xl">Cursor Card</h3>
      <p className="text-sm">
        This is a cursor card component.
      </p>
    </div>
  </CursorCard>
</CursorCardsContainer>
```

## Props


<Kbd>cursor-cards-container</Kbd> props.

| Prop           | Type                      | Description                                                 | Default   |
| -------------- | ------------------------- | ----------------------------------------------------------- | --------- |
| children       | <Kbd>React.ReactNode</Kbd> | The child components to render inside the container.        | -         |
| className      | <Kbd>string</Kbd>         | Additional custom classes.                                  | -         |
| proximityRange | <Kbd>number</Kbd>         | Range in pixels for detecting mouse proximity.             | 400       |


<Kbd>cursor-card</Kbd> props.

| Prop                | Type                      | Description                                                 | Default      |
| ------------------- | ------------------------- | ----------------------------------------------------------- | ------------ |
| children            | <Kbd>React.ReactNode</Kbd> | The content to render inside the card.                     | -            |
| className           | <Kbd>string</Kbd>         | Additional custom classes.                                  | -            |
| illuminationRadius  | <Kbd>number</Kbd>         | Radius of the illumination effect in pixels.               | 200          |
| illuminationColor   | <Kbd>string</Kbd>         | Color of the illumination effect.                          | "#FFFFFF10"  |
| illuminationOpacity | <Kbd>number</Kbd>         | Opacity of the illumination effect (0-1).                  | 0.8          |
| primaryHue          | <Kbd>string</Kbd>         | Primary color for the gradient border.                     | "#EC4899"    |
| secondaryHue        | <Kbd>string</Kbd>         | Secondary color for the gradient border.                   | "#6D77D5"    |
| borderColor         | <Kbd>string</Kbd>         | Default border color when not hovering.                    | "#E5E5E5"    |
