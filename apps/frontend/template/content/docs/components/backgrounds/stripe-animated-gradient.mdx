          ```jsx
          "use client";

          import React, { useEffect, useRef } from "react";
          import { Gradient } from "@/registry/components/backgrounds/stripe-shader";

          export function AnimatedGradient({
            color1 = "#a960ee",
            color2 = "#ff333d",
            color3 = "#90e0ff",
            color4 = "#ffcb57",
          }) {
            const canvasRef = useRef(null);

            useEffect(() => {
              const gradient = new Gradient();
              if (canvasRef.current) {
                gradient.initGradient(canvasRef.current);
              }
            }, [color1, color2, color3, color4]);

            return (
              <canvas
                ref={canvasRef}
                style={{
                  "--gradient-color-1": color1,
                  "--gradient-color-2": color2,
                  "--gradient-color-3": color3,
                  "--gradient-color-4": color4,
                }}
                className="inset-0 w-full h-full absolute z-0 rounded-[inherit]"
              />
            );
          }
          ```

## Usage

```tsx
import { AnimatedGradient } from "@/components/ui/stripe-animated-gradient";
```

```tsx
<AnimatedGradient
  color1="#a960ee"
  color2="#ff333d"
  color3="#90e0ff"
  color4="#ffcb57"
/>
```

> The parent container must be set to `position: relative` for the BorderBeam effect to work properly.  
> Additionally, this component is likely resource-intensive and will likely undergo a rework soon.

## Props

<Kbd>AnimatedGradient</Kbd> props.

| Prop   | Type   | Description                   | Default   |
| ------ | ------ | ----------------------------- | --------- |
| color1 | <Kbd>string</Kbd> | First color of the gradient.  | `#a960ee` |
| color2 | <Kbd>string</Kbd> | Second color of the gradient. | `#ff333d` |
| color3 | <Kbd>string</Kbd> | Third color of the gradient.  | `#90e0ff` |
| color4 | <Kbd>string</Kbd> | Fourth color of the gradient. | `#ffcb57` |
